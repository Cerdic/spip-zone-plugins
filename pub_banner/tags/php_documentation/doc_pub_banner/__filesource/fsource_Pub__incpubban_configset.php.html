  106. function pubban_recuperer_statut($url){    
  107.     include_spip('inc/distant');
  108.     // ouvrir la connexion et envoyer la requete et ses en-tetes
  109.     list($f, $fopen) = init_http('GET', $url, false, '', '');
  110.     if (!$f) return false;
  111.  
  112.     // Sauf en fopen, envoyer le flux d'entree
  113.     // et recuperer les en-tetes de reponses
  114.     if ($fopen)
  115.         $status = '';
  116.     else {
  117.         $s = @trim(fgets($f, 16384));
  118.         if (!preg_match(',^HTTP/[0-9]+\.[0-9]+ ([0-9]+),', $s, $r)) {
  119.             return 0;
  120.         }
  121.         $status = intval($r[1]);
  122.     }
  123.     return $status;
  124. }

  126. function pubban_enregistrer_config($args){
  127.     $mess = array();
  128.     if(!is_array($args)) return;
  129.     $_conf = pubban_recuperer_config();
  130.  
  131.     if($args['adds_ok']=='oui' AND strlen($args['adds_squelette'])){
  132.         $url_adds_skel = $GLOBALS['meta']['adresse_site'].'/?page='
  133.             .substr($args['adds_squelette'], 0, strpos($args['adds_squelette'], '.'));
  134.         $stat_adds_skel = pubban_recuperer_statut($url_adds_skel);
  135.         if(substr($stat_adds_skel,0,1) != '2') 
  136.             $mess['error'] = _T('pubban:erreur_url_statut', array('url'=>$url_adds_skel, 'statut'=>$stat_adds_skel));
  137.         else $_conf['adds_squelette'] = $url_adds_skel;
  138.     }
  139.     if(strlen($args['adds_commande'])){
  140.         $url_adds_comm = $GLOBALS['meta']['adresse_site'].'/?page='
  141.             .substr($args['adds_commande'], 0, strrpos($args['adds_commande'], '.'));
  142.         $stat_adds_comm = pubban_recuperer_statut($url_adds_comm);
  143.         if(substr($stat_adds_comm,0,1) != '2') 
  144.             $mess['error'] = _T('pubban:erreur_url_statut', array('url'=>$url_adds_comm, 'statut'=>$stat_adds_comm));
  145.         else $_conf['adds_commande'] = $url_adds_comm;
  146.     }
  147.     $conf = array_merge($_conf, $args);
  148.     include_spip('inc/meta');
  149.     ecrire_meta('pubban_config', serialize($conf), 'non');
  150.     ecrire_metas();
  151.     $mess['ok'] = _T('pubban:config_ok');
  152.     return $mess;
  153. }
