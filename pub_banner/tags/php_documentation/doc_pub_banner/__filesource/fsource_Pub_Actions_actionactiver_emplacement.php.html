  1. <?php
  2. /**
  3.  * Change le statut des emplacements
  4.  *
  5.  * Il s'agit d'une 'action SPIP' (<i>url_action()</i>) qui doit recevoir en argument une
  6.  * chaine construite sur le modèle : 'action-ID' avec 'action' :
  7.  * <ul>
  8.  * <li>'desactiver' : désactivation : passe en '1inactif' si était '2actif'</li>
  9.  * <li>'activer' : activation : passe en '2actif' si était '1inactif'</li>
  10.  * <li>'rompu' : introuvable : passe en '4rompu' si était '1inactif' ou '2actif'</li>
  11.  * <li>'out_trash' : sortie de poubelle : passe en '1inactif' si était '5poubelle'</li>
  12.  * <li>'trash' : poubelle : passe en '5poubelle' si était '1inactif' ou '2actif'</li>
  13.  * </ul>
  14.  *
  15.  * @name         Activer banniere
  16.  * @author         Piero Wbmstr <@link piero.wbmstr@gmail.com>
  17.  * @license        http://creativecommons.org/licenses/by-nc-sa/3.0/ Creative Commons BY-NC-SA
  18.  * @version     1.0 (06/2009)
  19.  * @package        Pub Banner
  20.  * @subpackage    Actions
  21.  */
  22. if (!defined("_ECRIRE_INC_VERSION")) return;
  23.  
  24. function action_activer_emplacement(){
  25.     include_spip('inc/autoriser');
  26.     include_spip('inc/pubban_process');
  27.     $securiser_action = charger_fonction('securiser_action', 'inc');
  28.     $arg = $securiser_action();
  29.     list($activer, $id_empl) = explode('-', $arg);
  30.  
  31.     if (intval($id_empl)){
  32.         $statut = pubban_recuperer_emplacement($id_empl, "statut");
  33.         switch ($activer) {
  34.             case 'activer' :
  35.                 if($statut == '1inactif') 
  36.                     $ok = pubban_editer_emplacement($id_empl, array("statut" => "2actif"));
  37.                 break;
  38.             case 'desactiver' :
  39.                 if($statut == '2actif') 
  40.                     $ok = pubban_editer_emplacement($id_empl, array("statut" => "1inactif"));
  41.                 break;
  42.             case 'rompu' :
  43.                 if(in_array($statut, array('1inactif', '2actif')))
  44.                     $ok = pubban_editer_emplacement($id_empl, array("statut" => "4rompu"));
  45.                 break;
  46.             case 'trash' :
  47.                 if(in_array($statut, array('1inactif', '2actif')))
  48.                     $ok = pubban_editer_emplacement($id_empl, array("statut" => "5poubelle"));
  49.                 break;
  50.             case 'out_trash' :
  51.                 if($statut == '5poubelle') 
  52.                     $ok = pubban_editer_emplacement($id_empl, array("statut" => "1inactif"));
  53.                 break;
  54.         }
  55.         if ($redirect = _request('redirect') ) {
  56.             $redirect = str_replace('&amp;', '&', $redirect);
  57.             if( $mode = _request('mode') )
  58.                 $redirect = parametre_url($redirect, 'mode', $mode);
  59.             include_spip('inc/headers');
  60.             redirige_par_entete( $redirect );
  61.         }
  62.     }
  63.  
  64.     return;
  65. }
  66. ?>
