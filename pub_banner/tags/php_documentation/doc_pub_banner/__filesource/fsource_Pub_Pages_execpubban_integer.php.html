  10. if (!defined("_ECRIRE_INC_VERSION")) return;

  12. function exec_pubban_integer_dist() {
  13.     global $connect_statut, $connect_id_auteur, $spip_lang_right;
  14.     if ($connect_statut != "0minirezo" ) { include_spip('inc/minipres'); echo minipres(); exit; }
  15.     include_spip('inc/presentation');
  16.     include_spip('inc/pubban_process');
  17.     $commencer_page = charger_fonction('commencer_page', 'inc');
  18.      $titre_page = _T('pubban:details_empl');
  19.     $rubrique = 'pubban';
  20.     $sous_rubrique = "home_pub";
  21.  
  22.     $id_empl = _request('id_empl');
  23.     $empl = pubban_recuperer_emplacement($id_empl);
  24.     $list_pub = pubban_pubs_de_emplacement($id_empl);
  25.     $s = ($empl['statut'] == '5poubelle') ? '3' : ( ($empl['statut'] == '1inactif') ? '2' : '1' );
  26.     $titre_empl = ( $id_empl != '0' ) ? "<div class='numero'>"._T('pubban:titre_info_empl')."<p>".$id_empl."</p>" : "<div class='numero'>"._T('pubban:titre_nouvel_empl') ;
  27.     $boite = "<div class='cadre_padding'><div class='infos'>"
  28.         .$titre_empl
  29.         ."<ul id='instituer_emplacement-".$id_empl."' class='instituer_emplacement instituer'><li>"._T('pubban:empl_is')."&nbsp;:"
  30.         ."<ul>"
  31.         ."<li class='publie ".(($s == '1') ? "selected" : "" )."'><a href='".generer_action_auteur("activer_emplacement", "activer-$id_empl", self('&'))."' onclick='return confirm(confirm_changer_statut);'><img src='../prive/images/puce-verte.gif' alt=\""._T('pubban:actif')."\"  />"._T('pubban:actif')."</a></li>"
  32.         ."<li class='prop ".(($s == '2') ? "selected" : "" )."'><a href='".generer_action_auteur("activer_emplacement", "desactiver-$id_empl", self('&'))."' onclick='return confirm(confirm_changer_statut);'><img src='../prive/images/puce-rouge.gif' alt=\""._T('pubban:inactif')."\"  />"._T('pubban:inactif')."</a></li>"
  33.         ."<li class='poubelle ".(($s == '3') ? "selected" : "" )."'><a href=\"javascript:delete_entry('".generer_action_auteur("activer_emplacement", "trash-$id_empl", generer_url_ecrire('pubban_integer_tous'))."', '"._T('pubban:confirm_delete_empl')."');\"><img src='../prive/images/puce-poubelle.gif' alt=\""._T('pubban:poubelle')."\"  />"._T('pubban:poubelle')."</a></li>"
  34.         ."</ul></li></ul>"
  35.         ."</div><div class='nettoyeur'></div></div></div>";
  36.  
  37.     $res = icone_horizontale(_T('pubban:home'), generer_url_ecrire("pubban_admin"), find_in_path("img/stock_home.png"), "rien.gif", false)
  38.         . icone_horizontale(_T('pubban:retour_liste_empl'), generer_url_ecrire("pubban_integer_tous"), find_in_path("img/stock_left.png"), "rien.gif", false)
  39.         . icone_horizontale(_T('pubban:nouveau_empl'), generer_url_ecrire('pubban_integer_edit','id_empl=new'), find_in_path("img/stock_insert-image.png"), "creer.gif", false)
  40.         . icone_horizontale(_T('pubban:page_stats'), generer_url_ecrire('pubban_stats'), find_in_path("img/stock-tool-button-color-balance.png"), "rien.gif", false);
  41.     $trash = pubban_poubelle_pleine();
  42.     if($trash) $res .= icone_horizontale(_T('pubban:open_trash'), generer_url_ecrire('pubban_admin','mode=trash'), find_in_path("img/stock_delete.png"), "rien.gif", false);
  43.  
  44.     $contexte = array(
  45.         'id_empl' => $id_empl,
  46.         'icone_editer' => icone_inline(_T('bouton_modifier'), generer_url_ecrire("pubban_integer_edit","id_empl=$id_empl"), find_in_path("img/stock_edit.png"), "rien.gif",$GLOBALS['spip_lang_right']),
  47.         'icone_nouvelle_pub' => icone_inline(_T('pubban:nouveau_pub_dans_emplacement'), generer_url_ecrire("pubban_pub_edit","id_pub=new&id_empl=$id_empl"), find_in_path("img/stock_insert-object.png"), "creer.gif",$GLOBALS['spip_lang_right']),
  48.         'redirect'=> generer_url_ecrire("pubban_integer"),
  49.         'btn_apercu' => $GLOBALS['pubban_btns']['apercu'],
  50.         'btn_editer' => $GLOBALS['pubban_btns']['editer'],
  51.         'btn_poubelle' => $GLOBALS['pubban_btns']['poubelle'],
  52.         'btn_inactive' => $GLOBALS['pubban_btns']['inactif'],
  53.         'btn_active' => $GLOBALS['pubban_btns']['actif'],
  54.         'btn_obsolete' => $GLOBALS['pubban_btns']['obsolete'],
  55.         'decompte_pub' => count($list_pub),
  56.         'decompte_pub_actives' => 0,
  57.         'decompte_pub_inactives' => 0,
  58.         'decompte_pub_obsoletes' => 0,
  59.         'listing_pub' => array(),
  60.     );
  61.     foreach($list_pub as $k=>$id_pub){
  62.         $statut = pubban_recuperer_pub($id_pub, 'statut');
  63.         $contexte['listing_pub'][] = $id_pub;
  64.         if($statut == '2actif') $contexte['decompte_pub_actives']++;
  65.         elseif($statut == '1inactif') $contexte['decompte_pub_inactives']++;
  66.         elseif($statut == '3obsolete') $contexte['decompte_pub_obsoletes']++;
  67.     }
  68.     $milieu = recuperer_fond("prive/emplacement_voir",$contexte);
  69.  
  70.     echo($commencer_page(_T('pubban:pubban')." - ".$titre_page, $rubrique, $sous_rubrique)), debut_gauche('', true),
  71.         "<br />" . debut_boite_info(true). $boite . fin_boite_info(true), bloc_des_raccourcis($res);
  72.     if(_request('retour')) echo icone_inline(_T('pubban:retour_search'), _request('retour'), find_in_path("img/stock_search.png"), "rien.gif", $GLOBALS['spip_lang_left']);
  73.     echo creer_colonne_droite('', true), debut_droite('', true), debut_cadre_trait_couleur(false, true),
  74.         pipeline('affiche_milieu', array( 'args' => array( 'exec' => 'pubban_integer' ), 'data' => $milieu ) ),
  75.         fin_cadre_trait_couleur(true), fin_gauche(), fin_page();
  76. }
