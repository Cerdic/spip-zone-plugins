  139. /**
  140.  * Recuperation des donnes d'une publicite
  141.  * @param    integer    $id_pub    L'ID de la pub a recuperer
  142.  * @param    string    $str    Le nom d'un paramètre à récupérer (optionnel)
  143.  * @return array    Les données de la pub (ou la valeur du paramètre si demandé)
  144.  */
  145. function pubban_recuperer_pub($id_pub, $str=false) {
  146.     $vals = array();
  147.     if($id_pub != '0') {
  148.         $resultat = sql_select("*", $GLOBALS['_PUBBAN_CONF']['table_pub'],"id_pub=".intval($id_pub)    , '', '', '', '', _BDD_PUBBAN);
  149.         if (sql_count($resultat) > 0) {
  150.             while ($row=spip_fetch_array($resultat)) {
  151.                 $vals['id'] = $id_pub;
  152.                 $vals['type'] = $row['type'];
  153.                 $vals['titre'] = $row['titre'];
  154.                 $vals['url'] = $row['url'];
  155.                 $vals['objet'] = $row['objet'];
  156.                 $vals['illimite'] = $row['illimite'];
  157.                 $vals['affichages'] = $row['affichages'];
  158.                 $vals['clics'] = $row['clics'];
  159.                 $vals['affichages_restant'] = $row['affichages_restant'];
  160.                 $vals['clics_restant'] = $row['clics_restant'];
  161.                 $vals['date_debut'] = $row['date_debut'];
  162.                 $vals['date_fin'] = $row['date_fin'];
  163.                 $vals['date_add'] = $row['date_add'];
  164.                 $vals['statut'] = $row['statut'];
  165.             }
  166.             sql_free($resultat, _BDD_PUBBAN);
  167.         }
  168.         $resultat_empl = sql_select("*", $GLOBALS['_PUBBAN_CONF']['table_join'],"id_pub=".intval($id_pub), '', '', '', '', _BDD_PUBBAN);
  169.         if (sql_count($resultat_empl) > 0) {
  170.             while ($row_empl=spip_fetch_array($resultat_empl)) {
  171.                 $vals['emplacement'][] = $row_empl['id_empl'];
  172.             }
  173.             sql_free($resultat_empl, _BDD_PUBBAN);
  174.         }
  175.     }
  176.     if($str){
  177.         if( isset($vals[$str]) ) return $vals[$str];
  178.         return false;
  179.     }
  180.     return $vals;
  181. }

  183. function pubban_comparer_emplacements($emp){
  184.     if(!is_array($emp)) return;
  185.     if(count($emp) > 1) {
  186.         $width = $height = array();
  187.         foreach($emp as $k=>$empl){
  188.             $width[] = pubban_recuperer_emplacement($empl, 'width');
  189.             $height[] = pubban_recuperer_emplacement($empl, 'height');
  190.         }
  191.         if( count(array_unique($width)) != 1 OR 
  192.             count(array_unique($height)) != 1
  193.         ) return false;
  194.     }
  195.     return true;
  196. }

  291. function pubban_calculer_prix($id_emp, $nb_aff){
  292.     $conf = pubban_recuperer_config();
  293.     $empl = pubban_recuperer_emplacement($id_emp);
  294.     if($nb_aff < $conf['tranche_1']) return $empl['prix_tranche_1'];
  295.     if($conf['tranche_1'] <= $nb_aff AND $nb_aff < $conf['tranche_2']) return $empl['prix_tranche_2'];
  296.     if($conf['tranche_2'] <= $nb_aff AND $nb_aff < $conf['tranche_3']) return $empl['prix_tranche_3'];
  297.     if($conf['tranche_3'] <= $nb_aff) return $empl['prix_tranche_4'];
  298.     return false;
  299. }

  445. function pubban_url_reponse($url) { 
  446.     $url = @parse_url($url);
  447.     if (!$url) return false;
  448.     $url = array_map('trim', $url);
  449.     $url['port'] = (!isset($url['port'])) ? 80 : (int)$url['port'];
  450.     $path = (isset($url['path'])) ? $url['path'] : '';
  451.     if ($path == '') $path = '/';
  452.     $path .= (isset($url['query'])) ? "?$url[query]" : '';
  453.     if (isset($url['host']) AND $url['host'] != gethostbyname($url['host'])){
  454.         if (PHP_VERSION >= 5)
  455.             $headers = get_headers("$url[scheme]://$url[host]:$url[port]$path");
  456.         else {
  457.             $fp = @fsockopen($url['host'], $url['port'], $errno, $errstr, 30);
  458.             if (!$fp) return false;
  459.             fputs($fp, "HEAD $path HTTP/1.1\r\nHost: $url[host]\r\n\r\n");
  460.             $headers = fread($fp, 4096);
  461.             fclose($fp);
  462.         }
  463.         $headers = (is_array($headers)) ? implode("\n", $headers) : $headers;
  464.         return (bool)preg_match('#^HTTP/.*\s+[(200|301|302)]+\s#i', $headers);
  465.     }
  466.     return false;
  467. }
