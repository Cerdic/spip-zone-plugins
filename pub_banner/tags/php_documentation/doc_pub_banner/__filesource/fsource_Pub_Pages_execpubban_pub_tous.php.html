  15.     include_spip('inc/presentation');

  18.     $rubrique = 'pubban';
  19.     $sous_rubrique = "home_pub";
  20.  
  21.     $res = icone_horizontale(_T('pubban:home'), generer_url_ecrire("pubban_admin"), find_in_path("img/stock_home.png"), "rien.gif", false)
  22.         . icone_horizontale(_T('pubban:nouveau_pub'), generer_url_ecrire('pubban_pub_edit','id_pub=new'), find_in_path("img/stock_insert-object.png"), "creer.gif", false)
  23.         . icone_horizontale(_T('pubban:page_stats'), generer_url_ecrire('pubban_stats'), find_in_path("img/stock-tool-button-color-balance.png"), "rien.gif", false);
  24.     $trash = pubban_poubelle_pleine();
  25.     if($trash) $res .= icone_horizontale(_T('pubban:open_trash'), generer_url_ecrire('pubban_admin','mode=trash'), find_in_path("img/stock_delete.png"), "rien.gif", false);
  26.     
  27.     $contexte = array(
  28.         'redirect'=> generer_url_ecrire("pubban_pub"),
  29.         'inverse' => _request('inverse'),
  30.         'inverse_2' => _request('inverse_2'),
  31.         'liste_2' => _request('inverse_2') ? _request('inverse_2') : _request('liste_2'),
  32.         'btn_apercu' => $GLOBALS['pubban_btns']['apercu'],
  33.         'btn_editer' => $GLOBALS['pubban_btns']['editer'],
  34.         'btn_poubelle' => $GLOBALS['pubban_btns']['poubelle'],
  35.         'btn_sortie_poubelle' => $GLOBALS['pubban_btns']['sortie_poubelle'],
  36.         'btn_inactif' => $GLOBALS['pubban_btns']['inactif'],
  37.         'btn_inactive' => $GLOBALS['pubban_btns']['inactif'],
  38.         'btn_active' => $GLOBALS['pubban_btns']['actif'],
  39.         'btn_obsolete' => $GLOBALS['pubban_btns']['obsolete'],
  40.     );

  42.  
  43.     echo($commencer_page(_T('pubban:pubban')." - ".$titre_page, $rubrique, $sous_rubrique)), debut_gauche('', true),
  44.         debut_cadre_relief(find_in_path("img/ico-pubban.png"), true, "", "<br />"._T('pubban:intro_pub')),

  47.         pipeline('affiche_milieu', array( 'args' => array( 'exec' => 'pubban_pub_tous' ), 'data' => $milieu ) ),
  48.         fin_gauche(), fin_page();
  49. }
