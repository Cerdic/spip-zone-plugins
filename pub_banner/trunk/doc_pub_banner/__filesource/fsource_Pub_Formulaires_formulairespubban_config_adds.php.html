  12. function formulaires_pubban_config_adds_charger_dist($serveurs='non'){
  13.     $valeurs = array(
  14.         'adds_ok' =>
  15.             _request('adds_ok') ? _request('adds_ok') : (
  16.                 _PUBBAN_ADDS ? 'oui' : 'non'
  17.             ),
  18.         'adds_squelette' => 
  19.             _request('adds_squelette') ? _request('adds_squelette') : (
  20.                 _PUBBAN_ADDS ? PUBBAN_SKEL_ADDS : $GLOBALS['_PUBBAN_CONF']['config']['adds_squelette']
  21.             ),
  22.         'id_article_adds' => 
  23.             _request('id_article_adds') ? _request('id_article_adds') : (
  24.                 _PUBBAN_ADDS ? PUBBAN_ID_ARTICLE_ADDS : $GLOBALS['_PUBBAN_CONF']['config']['id_article_adds']
  25.             ),
  26.         'adds_commande' => 
  27.             _request('adds_commande') ? _request('adds_commande') : (
  28.                 defined('PUBBAN_SKEL_COMMANDE') ? PUBBAN_SKEL_COMMANDE : ''
  29.             ),
  30.         'id_article_adds_commande' => 
  31.             _request('id_article_adds_commande') ? _request('id_article_adds_commande') : (
  32.                 defined('PUBBAN_ID_ARTICLE_COMMANDE') ? PUBBAN_ID_ARTICLE_COMMANDE : ''
  33.             ),
  34.     );
  35.     return $valeurs;
  36. }

  38. function formulaires_pubban_config_adds_verifier_dist(){
  39.     global $erreurs;
  40.     if(_request('adds_ok')=='oui'){
  41.         if(!_request('adds_squelette')) $erreurs['adds_squelette'] = _T('pubban:erreur_adds_squelette');
  42.         if(!_request('adds_commande') AND _request('id_article_adds_commande')=='null')
  43.             $erreurs['adds_commande'] = _T('pubban:erreur_adds_commande');
  44.     }
  45.  
  46.     return $erreurs;
  47. }
  48.  
  49. function formulaires_pubban_config_adds_traiter_dist(){
  50.     $datas = array(
  51.         'adds_ok' => ($adds = _request('adds_ok') AND $adds == 'oui' ) ? 'oui' : 'non',
  52.         'adds_squelette' => _request('adds_squelette'),
  53.         'id_article_adds' => _request('id_article_adds'),
  54.         'adds_commande' => _request('adds_commande'),
  55.         'id_article_adds_commande' => _request('id_article_adds_commande'),
  56.     );
  57.  
  58.     $do = pubban_enregistrer_config($datas);
  59.     return array(
  60.         'message_ok' => $do['ok'],
  61.         'message_erreur' => $do['error']
  62.     );
  63. }
