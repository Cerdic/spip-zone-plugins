  10. if (!defined("_ECRIRE_INC_VERSION")) return;

  12. function exec_pubban_admin_dist() {
  13.     global $connect_statut, $connect_id_auteur, $spip_lang_right;
  14.     include_spip('inc/presentation');
  15.     $commencer_page = charger_fonction('commencer_page', 'inc');
  16.      $titre_page = _T('pubban:gestion_pubban');
  17.     $rubrique = 'pubban';
  18.     $bloc_racc = $bloc_stats = '';
  19.     $texte_page = _T("pubban:texte_admin");
  20.  
  21.     $boutons = "<br class='nettoyeur' />"
  22.         . icone_inline(_T('pubban:page_stats'), generer_url_ecrire('pubban_stats'), find_in_path("img/stock-tool-button-color-balance.png"), "rien.gif", $GLOBALS['spip_lang_right'])
  23.         . icone_inline(_T('pubban:page_infos'), generer_url_ecrire('pubban_info'), find_in_path("img/status-dock-24.png"), "rien.gif", $GLOBALS['spip_lang_right']);
  24.  
  25.     $search_str = _request('search_pubban');
  26.     $contexte = array(
  27.         'search_pubban' => $search_str,
  28.         'redirect'=> generer_url_ecrire("pubban_admin"),
  29.         'inverse' => _request('inverse'),
  30.         'btn_apercu' => $GLOBALS['pubban_btns']['apercu'],
  31.         'btn_editer' => $GLOBALS['pubban_btns']['editer'],
  32.         'btn_poubelle' => $GLOBALS['pubban_btns']['poubelle'],
  33.         'btn_inactive' => $GLOBALS['pubban_btns']['inactif'],
  34.         'btn_active' => $GLOBALS['pubban_btns']['actif'],
  35.         'btn_obsolete' => $GLOBALS['pubban_btns']['obsolete'],
  36.         'btn_sortie_poubelle' => $GLOBALS['pubban_btns']['sortie_poubelle'],
  37.         'puce_empl' => $GLOBALS['_PUBBAN_PUCES_STATUTS']['emplacement']['icon'],
  38.     );
  39.  
  40.     $milieu = recuperer_fond("prive/search", $contexte);
  41.     $contenu = pipeline('affiche_milieu',array('args'=>array('exec'=>'pubban_admin'),'data'=>$milieu));
  42.  
  43.     if ($connect_statut == "0minirezo" ) {
  44.         $trash = pubban_poubelle_pleine();
  45.         $boutons .= icone_inline(_T('pubban:liste_pub'), generer_url_ecrire('pubban_pub_tous'), find_in_path("img/stock_insert-object.png"), "rien.gif", $GLOBALS['spip_lang_right'])
  46.             . icone_inline(_T('pubban:list_empl'), generer_url_ecrire('pubban_integer_tous'), find_in_path("img/stock_insert-image.png"), "rien.gif", $GLOBALS['spip_lang_right'])
  47.             . icone_inline(_T('pubban:page_config'), generer_url_ecrire('pubban_config'), find_in_path("img/stock_import.png"), "rien.gif", $GLOBALS['spip_lang_right'])
  48.             . ( _PUBBAN_ADDS ? icone_inline(_T('pubban:page_tarifs'), generer_url_ecrire('pubban_tarifs'), find_in_path("img/money.png"), "rien.gif", $GLOBALS['spip_lang_right']) : '')
  49.             . ( $trash ? icone_inline(_T('pubban:open_trash'), generer_url_ecrire('pubban_admin','mode=trash'), find_in_path("img/stock_delete.png"), "rien.gif", $GLOBALS['spip_lang_right']) : '');
  50.         $res = icone_horizontale(_T('pubban:nouveau_pub'), generer_url_ecrire('pubban_pub_edit','id_pub=new'), find_in_path("img/stock_insert-object.png"), "creer.gif", false)
  51.             . icone_horizontale(_T('pubban:nouveau_empl'), generer_url_ecrire('pubban_integer_edit','id_empl=new'), find_in_path("img/stock_insert-image.png"), "creer.gif", false);
  52.         $bloc_racc = bloc_des_raccourcis($res);
  53.  
  54.         //poubelle
  55.         if( _request('mode') == 'trash' ){
  56.             $stats_ferme = true;
  57.             $milieu = recuperer_fond("prive/pub_liste_trash",$contexte);
  58.             $contenu_add = pipeline('affiche_milieu',array('args'=>array('exec'=>'pubban_admin'),'data'=>$milieu))
  59.                 . icone_inline(_T('pubban:vider_trash'), "javascript:delete_entry(\"".generer_action_auteur('vider_poubelle_pubban', false, generer_url_ecrire('pubban_admin','mode=trash'))."\", \""._T('pubban:confirm_vider_poubelle')."\");", find_in_path("img/trash-empty-accept.png"), "rien.gif", $GLOBALS['spip_lang_right'])
  60.                 . icone_inline(_T('pubban:home'), generer_url_ecrire("pubban_admin"), find_in_path("img/stock_home.png"), "rien.gif", $GLOBALS['spip_lang_right']);
  61.         }
  62.         //recherche
  63.         elseif( strlen($search_str) ){
  64.             $stats_ferme = true;
  65.             $search = pubban_search(_request('search_pubban'));
  66.             $contexte['pub'] = $search['pub'];
  67.             $contexte['emp'] = $search['emp'];
  68.             $contexte['boucle_search'] = count($contexte['pub'])+count($contexte['emp']);
  69.             $milieu = recuperer_fond("prive/search_results", $contexte);
  70.             $contenu_add = pipeline('affiche_milieu',array('args'=>array('exec'=>'pubban_admin'),'data'=>$milieu))
  71.                 . icone_inline(_T('pubban:home'), generer_url_ecrire("pubban_admin"), find_in_path("img/stock_home.png"), "rien.gif", $GLOBALS['spip_lang_right']);
  72.         }
  73.     }
  74.  
  75.     $verif_empl = sql_countsel($GLOBALS['_PUBBAN_CONF']['table_empl'], '', '', '', '', '', _BDD_PUBBAN);
  76.     $verif = sql_countsel($GLOBALS['_PUBBAN_CONF']['table_pub'], '', '', '', '', '', _BDD_PUBBAN);
  77.     $bloc_pliable_plie = ( $verif != 0 AND $verif_empl != 0 AND !$stats_ferme ) ? 'deplie' : 'replie';
  78.     $bloc_pliable_titre = "<div class='titrem $bloc_pliable_plie' onmouseover=\"jQuery(this).depliant('#stats_div');\">"
  79.         ."<a href='#' onclick=\"return jQuery(this).depliant_clicancre('#stats_div');\" class='titremancre'></a>"
  80.         ."<strong>"._T('pubban:info_stats')."</strong></div>";
  81.     $bloc_stats .= debut_cadre_trait_couleur("rien.gif", true, '', $bloc_pliable_titre)
  82.         . afficher_statistiques_pubban(true, 'stats_div', $bloc_pliable_plie)
  83.         . fin_cadre_trait_couleur(true);
  84.  
  85.     echo($commencer_page(_T('pubban:pubban')." - ".$titre_page, $rubrique, '')), debut_gauche('', true), 
  86.         debut_cadre_relief(find_in_path("img/ico-pubban.png"), true, "", "<br />".$titre_page), 
  87.         $texte_page, pubban_lien_doc(), fin_cadre_relief(true), $bloc_racc, pubban_boite_info(),
  88.         creer_colonne_droite('', true), debut_droite('', true), gros_titre($titre_page, '', false),
  89.         $contenu, $bloc_stats, $contenu_add, ($contenu_add ? '' : $boutons), 
  90.         fin_gauche(), fin_page();
  91. }
