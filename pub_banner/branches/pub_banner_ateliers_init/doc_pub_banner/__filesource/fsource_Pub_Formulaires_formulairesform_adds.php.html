  27. function formulaires_form_adds_charger_dist($retour='', $lier_trad = 0){
  28.     global $list_emp;
  29.     // on bloque l'acces si pas 'ads_ok' dans la config
  30.     if( $adds = pubban_recuperer_config('adds_ok') AND $adds == 'non') {
  31.         return array('adds_off' => 'oui');
  32.     }
  33.     // on recupere l'eventuel emplacement passe en URL
  34.     $id_empl_url = _request('id_empl');
  35.     $valeurs = array(
  36.         'titre' => _request('titre') ? _request('titre') : '',
  37.         'url' => _request('url') ? _request('url') : '',
  38.         'code' => _request('code') ? _request('code') : '',
  39.         'type' => _request('type') ? _request('type') : 'img',
  40.         'type_swf' => _request('type_swf') ? _request('type_swf') : 'non',
  41.         'type_flash' => _request('type_flash') ? _request('type_flash') : 'non',
  42.         'nb_aff' => _request('nb_aff') ? _request('nb_aff') : false,
  43.         'global_tranche_1' => pubban_recuperer_config('tranche_1'),
  44.         'global_tranche_2' => pubban_recuperer_config('tranche_2'),
  45.         'global_tranche_3' => pubban_recuperer_config('tranche_3'),
  46.         'global_tranche_1_prix' => pubban_recuperer_config('prix_tranche_1'),
  47.         'global_tranche_2_prix' => pubban_recuperer_config('prix_tranche_2'),
  48.         'global_tranche_3_prix' => pubban_recuperer_config('prix_tranche_3'),
  49.         'global_tranche_4_prix' => pubban_recuperer_config('prix_tranche_4'),
  50.         'list_emplacements' => $list_emp,
  51.      );
  52.     foreach($list_emp as $k=>$id_empl){
  53.         $valeurs["empl_$id_empl"] = ($id_empl == $id_empl_url) ? 'oui' : 'non';
  54.     }
  55.     return($valeurs);
  56. }

  58. function formulaires_form_adds_verifier_dist($retour='', $lier_trad = 0){
  59.     $erreurs = array();
  60.     global $list_emp;
  61.     $previsu_ok = _request('previsu_ok') ? _request('previsu_ok') : false;
  62.  
  63.     $list_emp_sel = array();
  64.     foreach($list_emp as $k=>$id_empl){
  65.         if($id_sel = _request("empl_$id_empl") AND $id_sel == 'oui')
  66.             array_push(&$list_emp_sel, $id_empl);
  67.     }
  68.     if(!count($list_emp_sel)) 
  69.         $erreurs['emplacements'] = _T('pubban:erreur_empl');
  70.     if(!$titre = _request('titre') OR !strlen($titre)) 
  71.         $erreurs['titre'] = _T('pubban:erreur_titre');
  72.     if(!$url = _request('url') OR !strlen($url)) 
  73.         $erreurs['url'] = _T('pubban:erreur_url');
  74.     elseif(!pubban_UrlOK($url)) 
  75.         $erreurs['url'] = _T('pubban:erreur_url_not_url');
  76.     if(!$affs = _request('nb_aff')) 
  77.         $erreurs['nb_aff'] = _T('pubban:erreur_nb_aff');
  78.     if(!$code = _request('code')) 
  79.         $erreurs['code'] = _T('pubban:erreur_code');
  80.     elseif($type = _request('type') AND $type != 'flash' AND !pubban_UrlOK($code)) 
  81.         $erreurs['code'] = _T('pubban:erreur_url_not_url');
  82.     if( !count($erreurs) AND ( !$previsu_ok OR $previsu_ok != 'valide') ){
  83.         $erreurs['previsu'] = ' ';
  84.         $erreurs_secondaires = array();
  85.         if( !$ok = pubban_comparer_emplacements($list_emp_sel) ) 
  86.             $erreurs_secondaires['emplacements'] = _T('pubban:attention_emplacements_differents');
  87.         if($type != 'flash' AND !pubban_url_reponse($code)) 
  88.             $erreurs_secondaires['code'] = _T('pubban:erreur_img_not_url');
  89.         if( count($erreurs_secondaires) ) {
  90.             $erreurs['general'] = _T('pubban:voir_erreurs_ci_dessous');
  91.             $erreurs['previsu_ok'] = ( $previsu_ok == 'a_verifier' ) ? 'valide' : 'a_verifier';
  92.         }
  93.         else $erreurs['previsu_ok'] = 'valide';
  94.         $erreurs = array_merge($erreurs, $erreurs_secondaires);
  95.     }
  96.     return $erreurs;
  97. }

  99. function formulaires_form_adds_traiter_dist($retour='', $lier_trad = 0){
  100.     global $list_emp;
  101.     $list_emp_sel = array();
  102.     foreach($list_emp as $k=>$id_empl){
  103.         if($id_sel = _request("empl_$id_empl") AND $id_sel == 'oui')
  104.             array_push(&$list_emp_sel, $id_empl);
  105.     }
  106.     $affichages = _request('nb_aff');
  107.     $datas = array(
  108.         'titre' => _request('titre'),
  109.         'url' => _request('url'),
  110.         'objet' => _request('code'),
  111.         'affichages_restant' => pubban_transformer_nombre( $affichages ),
  112.         'type' => (_request('type_falsh') == 'oui') ? 'flash' : 
  113.             ( (_request('type_swf') == 'oui') ? 'swf' : 'img' ),
  114.     );
  115.     include_spip('inc/pubban_process');
  116.     if( list($panier, $id_pub) = pubban_adds_construire_panier($datas, $list_emp_sel) ){
  117.         $comm = pubban_adds_process_commande($panier, $id_pub);
  118.     }
  119.     return array(
  120.         'message_erreur' => _T('pubban:error_global')
  121.     );
  122. }
