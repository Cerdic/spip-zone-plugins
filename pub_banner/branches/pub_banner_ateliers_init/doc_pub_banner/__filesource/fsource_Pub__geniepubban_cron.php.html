  9. if (!defined("_ECRIRE_INC_VERSION")) return;
  10.  
  11. /**
  12.  * Action CRON
  13.  * Verifie la validite de chaque pub (selon ses dates) et active ou desactive
  14.  * si necessaire ...
  15.  */
  16. function genie_pubban_cron($time){
  17.     $nb_modif = $nb_delete = $nb_errors = 0;
  18.     $gdate = date("Y-m-d");
  19.  
  20.     $requete = sql_select("*", $GLOBALS['_PUBBAN_CONF']['table_pub'], "date_debut!='' OR date_fin!=''", '', '', '', '',_BDD_PUBBAN);
  21.     while($tableau = spip_fetch_array($requete)) {
  22.         $nom = $tableau['titre'];
  23.         $actif = $tableau['statut'];
  24.         $datedebut = (isset($tableau['date_debut']) AND strlen($tableau['date_debut'])) ? $tableau['date_debut'] : false;
  25.         $datefin = (isset($tableau['date_fin']) AND strlen($tableau['date_fin'])) ? $tableau['date_fin'] : false;
  26.         if($datedebut AND $actif == '1inactif' AND $datedebut >= $gdate) {
  27.             sql_updateq($GLOBALS['_PUBBAN_CONF']['table_pub'], array('statut' => '2actif'), "titre='".$nom."'", '', _BDD_PUBBAN);
  28.             $nb_modif++;
  29.         }
  30.         if($datefin AND $actif == '2actif' AND $datefin <= $gdate) {
  31.             sql_updateq($GLOBALS['_PUBBAN_CONF']['table_pub'], array( 'statut' => '3obsolete', 'affichages_restant' => '0', 'clics_restant' => '0' ), "titre='".$nom."'", '', _BDD_PUBBAN);
  32.             $nb_modif++;
  33.         }
  34.     }
  35.  
  36.     $requete2 = sql_select("id_pub, date_add", $GLOBALS['_PUBBAN_CONF']['table_pub'], "statut='0cree'", '', '', '', '',_BDD_PUBBAN);
  37.     while($tableau = spip_fetch_array($requete2)) {
  38.         list($date1, $rien) = explode(' ', $tableau['date_add']);
  39.         list($y1, $m1, $d1) = explode('-', $date1);
  40.         $time1 = mktime(0, 0, 0, $m1, $d1, $y1);
  41.         list($y2, $m2, $d2) = explode('-', $gdate);
  42.         $time2 = mktime(0, 0, 0, $m2, $d2, $y2);
  43.         if( ($time2-$time1) >= 2592000){
  44.             echo "ON y va pour id : ".$tableau['id_pub'];
  45.             if( $ok = sql_delete($GLOBALS['_PUBBAN_CONF']['table_pub'], 'id_pub='.$tableau['id_pub'], _BDD_PUBBAN) )
  46.                 $nb_delete++;
  47.             else $nb_errors++;
  48.         }
  49.     }
  50.  
  51.     $old_delete_stats = date('z', time() - 8726400);
  52.     sql_delete($GLOBALS['_PUBBAN_CONF']['table_stats'], "jour < ('".$old_delete_stats."')", _BDD_PUBBAN);
  53.  
  54.     spip_log("PUB BANNER CRON - Mise a jour OK [ $nb_modif updates | $nb_delete deletes | $nb_errors erreurs ]");
  55. }
  56. ?>
