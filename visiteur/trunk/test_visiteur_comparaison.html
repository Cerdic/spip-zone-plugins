Comparaison avec ==

  • visiteur nom == JLuc : #_VISITEUR{nom,==,JLuc}
  • visiteur nom == Anna : #_VISITEUR{nom,==,Anna}
  • visiteur nom != JLuc : #_VISITEUR{nom,!=,JLuc}
  • visiteur nom > J : #_VISITEUR{nom,>,J}
  • visiteur nom < J : #_VISITEUR{nom,<,J}
  • visiteur nom > K : #_VISITEUR{nom,>,K}
  • visiteur nom < K : #_VISITEUR{nom,<,K}
