Comparaison avec ==

Le nom est : #_SESSION{nom}
  • _session nom == JLuc : #_SESSION{nom,==,JLuc}
  • _session nom == G : #_SESSION{nom,==,G}
  • _session nom != JLuc : #_SESSION{nom,!=,JLuc}
  • _session nom != G : #_SESSION{nom,!=,G}
  • _session nom > F : #_SESSION{nom,>,F}
  • _session nom < F : #_SESSION{nom,<,F}
  • _session nom > G : #_SESSION{nom,>,G}
  • _session nom < G : #_SESSION{nom,<,G}
  • _session nom > J : #_SESSION{nom,>,J}
  • _session nom < J : #_SESSION{nom,<,J}
  • _session nom > K : #_SESSION{nom,>,K}
  • _session nom < K : #_SESSION{nom,<,K}
