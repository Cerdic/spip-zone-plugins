# _AUTORISER_SI avec valeurs calculées d'id - DANS UNE BOUCLE

Environnement reçu # ID_ARTICLE=#ID_ARTICLE
Pour tester ce squelette il faut debug=oui dans l'url
et avoir activé le debug dans macrosession_options.php

Dans la boucle # ID_ARTICLE=#ID_ARTICLE

Avec env : #_AUTORISER_SI{modifier,article,env} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN
Avec boucle : #_AUTORISER_SI{modifier,article,boucle} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN
Avec debug : #_AUTORISER_SI{modifier,article,debug} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN

Dans une boucle ARTICLES avec env

#_AUTORISER_SI{modifier,article,env} Autorisation accordée pour article #ID_ARTICLE #_AUTORISER_FIN

Dans une boucle DOCUMENTS avec env

#_AUTORISER_SI{modifier,document,env} Autorisation accordée pour DOCUMENT #ID_DOCUMENT #_AUTORISER_FIN

Dans une boucle ARTICLES avec boucle

#_AUTORISER_SI{modifier,article,boucle} Autorisation accordée pour article #ID_ARTICLE #_AUTORISER_FIN

Dans une boucle DOCUMENTS avec boucle

#_AUTORISER_SI{modifier,document,boucle} Autorisation accordée pour DOCUMENT #ID_DOCUMENT #_AUTORISER_FIN
