#SQUELETTE

Appel de _SESSION avec champ de session étendue

nom : #_SESSION{nom}

champ étendu : nbreste

nbreste
#_SESSION{nbreste}

avec filtres

nbreste |+ 3
#_SESSION{nbreste,plus,3}
nbreste |choixsivide vide! pasvide
#_SESSION{nbreste,choixsivide,vide!,pasvide}
nbreste |? pasvide! vide
#_SESSION{nbreste,?,pas vide!,vide}
