#SQUELETTE
Pour tester ce squelette on peut ajouter debug=oui dans l'url pour voir plus d'infos
Environnement reçu # ID_ARTICLE=#ID_ARTICLE
Ajouter id_article=1 à l'url

# _AUTORISER_SI avec valeurs calculées d'id - dans une BOUCLE


\#_AUTORISER_SI{modifier,article,...} dans BOUCLE(ARTICLE){id_article=1}


Avec \#ID_ARTICLE = #ID_ARTICLE : #_AUTORISER_SI{modifier,article,#ID_ARTICLE} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN
Avec env : #_AUTORISER_SI{modifier,article,env} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN
Avec boucle : #_AUTORISER_SI{modifier,article,boucle} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN
Avec debug : #_AUTORISER_SI{modifier,article,debug} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN

Dans une boucle (ARTICLES){0,3} avec env

#_AUTORISER_SI{modifier,article,env} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN pour article #ID_ARTICLE

Dans une boucle (DOCUMENTS){0,3} avec env

#_AUTORISER_SI{modifier,document,env} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN pour doc #ID_DOCUMENT

Dans une boucle ARTICLES avec boucle

#_AUTORISER_SI{modifier,article,boucle} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN pour article #ID_ARTICLE

Dans une boucle DOCUMENTS avec boucle

#_AUTORISER_SI{modifier,document,boucle} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN pour doc #ID_DOCUMENT
