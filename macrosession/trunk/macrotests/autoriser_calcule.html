Pour tester ce squelette il faut debug=oui dans l'url
et avoir activé le debug dans macrosession_options.php
Et l'article dont l'id_article est en url doit exister.
#SQUELETTE

# _AUTORISER_SI avec valeurs calculées d'id - HORS BOUCLE


Ici id_article=#ID_ARTICLE venu de l'env

_SESSION :
nom=#_SESSION{nom}
statut=#_SESSION{statut}

_AUTORISER:
#_AUTORISER_SI{modifier,article,1} OUI _AUTORISER modifier article 1 #_AUTORISER_SINON NON _AUTORISER modifier article 1 #_AUTORISER_FIN

Avec url : _AUTORISER_SI{modifier,article,url}

_AUTORISER modifier article url avec val de id_article : [(#VAL{id_article}|_request)] #_AUTORISER_SI{modifier,article,url} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN

Avec env : AUTORISER_SI{modifier,article,env}

Val d'env de id_article : #ID_ARTICLE #_AUTORISER_SI{modifier,article,env} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN

avec \#ID_ARTICLE : _AUTORISER_SI{modifier,article,\#ID_ARTICLE}

Val de \#ID_ARTICLE : #ID_ARTICLE #_AUTORISER_SI{modifier,article,#ID_ARTICLE} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN #SET{id_article,2}

Avec un #GET : \#_AUTORISER_SI{modifier,article,#GET{id_article}}

Aprés un \#SET, # GET{id_article} vaut : #GET{id_article}
#_AUTORISER_SI{modifier,article,#GET{id_article}} OUI #_AUTORISER_SINON NON #_AUTORISER_FIN
