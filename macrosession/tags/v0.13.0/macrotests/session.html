\#_SESSION sans extension ni filtre

  • _session : #_SESSION
  • _session nom : #_SESSION{nom}
  • _session statut : #_SESSION{statut}
  • _session email : #_SESSION{email}

Filtres sans arguments

  • _session{nom,strlen}:#_SESSION{nom,strlen}
  • _session{nom, strlen } avec espaces à strlen : #_SESSION{nom, strlen }
  • _session{ nom,strlen} avec espaces avant nom : #_SESSION{ nom,strlen}
  • _session{nom ,strlen} avec espaces après nom : #_SESSION{nom ,strlen}
  • _session{nom,filtre_inexistant} : #_SESSION{nom,filtre_inexistant}
  • _session{nom,mb_strtoupper}:#_SESSION{nom,mb_strtoupper}

Filtres avec argument constant

  • _session{nom,concat,_plof}:#_SESSION{nom,concat,_plof}
